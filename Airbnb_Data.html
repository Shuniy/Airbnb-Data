def clean_price(x):
    """
    clean the price column by eliminate the $ and , 
    from the string and return a float to be stored in the column
    """
    try:
        if ',' in x:
            x = x.replace(',','')
        x = x[1:]
        return float(str(x))
    except:
        return x

def days_between_dates(date):
    """
    return the number of days from today to the given date
    """
    try:
        date_format = "%Y-%m-%d"
        current = datetime.now()
        da = datetime.strptime(date, date_format)
        return abs((current - da).days)
    except:
        return date

def create_dummy_variables(df, cat_cols, dummy_na):
    '''
    INPUT:
    df - pandas dataframe with categorical variables you want to dummy
    cat_cols - list of strings that are associated with names of the categorical columns
    dummy_na - Bool holding whether you want to dummy NA vals of categorical columns or not
    
    OUTPUT:
    df - a new dataframe that has the following characteristics:
            1. contains all columns that were not specified as categorical
            2. removes all the original columns in cat_cols
            3. dummy columns for each of the categorical columns in cat_cols
            4. if dummy_na is True - it also contains dummy columns for the NaN values
            5. Use a prefix of the column name with an underscore (_) for separating 
    '''
    for col in cat_cols:
        try:
            df = pd.concat([df.drop(col, axis =1), pd.get_dummies(df[col], prefix=col, prefix_sep='_', drop_first=True, dummy_na=dummy_na)], axis=1)
        except:
            continue
    return df

    """
    Perform k-mean clustering on provided data and return the score for the given clusters 
    num
    """
    # Instantiate K-Means
    kmeans = KMeans(cluster)
    
    #Fitting model
    model = kmeans.fit(data)

    score = np.abs(model.score(data))
